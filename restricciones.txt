Iterator<Socio> it = f.getSocios().iterator();
DefaultTableModel mdl = (DefaultTableModel) tFamiliares.getModel();
mdl.setRowCount(0);
while (it.hasNext()) {
	Socio s = it.next();
	if (s.isVigente()) {  
		Object[] fila = new Object[4];
		fila[0] = s;
		fila[1] = s.getApellido();
		fila[2] = s.getFechaNac();
		fila[3] = s.getRol();
		mdl.addRow(fila); 
	}
}
